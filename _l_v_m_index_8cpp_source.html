46 bool Order::valid() const {
47  // Check that all indices are in range.
48  for (int i : order) {
49  if (i < 0 || i >= int(order.size())) {
50  return false;
51  }
52  }
53  // Check that there are no duplicates.
54  for (std::size_t i = 0; i < order.size(); i++) {
55  for (std::size_t j = i + 1; j < order.size(); j++) {
56  if (order[i] == order[j]) {
57  return false;
58  }
59  }
60  }
61  return true;
62 }

171  class Source : public Stream::Source {
172  const Order& order;
173 
174  // the begin and end of the stream
175  using iter = typename Structure::iterator;
176  iter cur;
177  iter end;
178 
179  // an internal buffer for re-ordered elements

237  Stream range(const TupleRef& low, const TupleRef& high) const override {
238  Entry a = order.encode(low.asTuple<Arity>());
239  Entry b = order.encode(high.asTuple<Arity>());
240  // Transfer upper_bound to a equivalent lower bound
241  bool fullIndexSearch = true;
242  for (size_t i = Arity; i-- > 0;) {
243  if (a[i] == MIN_RAM_DOMAIN && b[i] == MAX_RAM_DOMAIN) {
244  b[i] = MIN_RAM_DOMAIN;
245  continue;
246  }
247  if (a[i] == b[i]) {
248  b[i] += 1;
249  fullIndexSearch = false;
250  break;
251  }
252  }
253  assert(fullIndexSearch == false && "Full index search is not allowed in range query\n");
254  return std::make_unique<Source>(order, data.lower_bound(a), data.lower_bound(b));
255  }
256 
257  void clear() override {
258  data.clear();
259  }
