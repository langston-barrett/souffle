35  /** Encode a relation into a index Id and return the encoding result. */
36  size_t encodeRelation(const RamRelation& rel) {
37  const std::string& relationName = rel.getName();
38  auto iter = relNameToIndex.find(relationName);
39  // Create and give relation a new index if it is not in the environment yet
40  if (iter == relNameToIndex.end()) {
41  relNameToIndex.insert(std::make_pair(relationName, relNameToIndex.size()));
42  relationMap.push_back(createRelation(rel));
43  return relNameToIndex.size() - 1;
44  } else {
45  return iter->second;
46  }
47  }

78  /** Create relation with corresponding index type */
79  std::unique_ptr<LVMRelation> createRelation(const RamRelation& rel) {
80  const MinIndexSelection& orderSet = isa->getIndexes(rel);
81 
82  if (rel.getArity() > MAX_DIRECT_INDEX_SIZE) {
83  return std::make_unique<LVMIndirectRelation>(
84  rel.getArity(), rel.getName(), rel.getAttributeTypeQualifiers(), orderSet);
85  }
86 
87  switch (rel.getRepresentation()) {
88  case RelationRepresentation::BTREE:
89  return std::make_unique<LVMRelation>(
90  rel.getArity(), rel.getName(), rel.getAttributeTypeQualifiers(), orderSet);
91  case RelationRepresentation::BRIE:
92  return std::make_unique<LVMRelation>(rel.getArity(), rel.getName(),
93  rel.getAttributeTypeQualifiers(), orderSet, createBrieIndex);
94  case RelationRepresentation::EQREL:

97  case RelationRepresentation::DEFAULT:
98  return std::make_unique<LVMRelation>(
99  rel.getArity(), rel.getName(), rel.getAttributeTypeQualifiers(), orderSet);
100  default:
101  break;
102  }
103  return nullptr;
104  }

325  auto arity = exists.getRelation().getArity();
326  auto relId = relationEncoder.encodeRelation(exists.getRelation());
327  std::vector<int> typeMask(arity);
328  bool emptinessCheck = true;
329  bool fullExistenceCheck = true;

333  emptinessCheck = false;
334  typeMask[i] = 1;

336  fullExistenceCheck = false;
