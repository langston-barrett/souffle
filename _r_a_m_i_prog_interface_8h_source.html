172  * Implementation of SouffleProgram interface for an interpreter instance

174 class RAMIProgInterface : public SouffleProgram {
175 public:
176  RAMIProgInterface(RAMIInterface& interp)
177  : prog(*interp.getTranslationUnit().getProgram()), exec(interp),
178  symTable(interp.getTranslationUnit().getSymbolTable()) {
179  uint32_t id = 0;
180 
181  // Retrieve AST Relations and store them in a map
182  std::map<std::string, const RamRelation*> map;
183  visitDepthFirst(prog, [&](const RamRelation& rel) { map[rel.getName()] = &rel; });
184 
185  // Build wrapper relations for Souffle's interface
186  for (auto& rel_pair : exec.getRelationMap()) {
187  auto& name = rel_pair.first;
188  auto& interpreterRel = *rel_pair.second;
189  assert(map[name]);
190  const RamRelation& rel = *map[name];
191 
192  // construct types and names vectors
193  std::vector<std::string> types;
194  std::vector<std::string> attrNames;
195  for (size_t i = 0; i < rel.getArity(); i++) {
196  std::string t = rel.getArgTypeQualifier(i);
197  types.push_back(t);
198 
199  std::string n = rel.getArg(i);
200  attrNames.push_back(n);
201  }
202  auto* interface =
203  new RAMIRelInterface(interpreterRel, symTable, rel.getName(), types, attrNames, id);
204  interfaces.push_back(interface);
205  bool input;
206  bool output;
207  visitDepthFirst(prog, [&](const RamStore& store) {
208  if (store.getRelation() == rel) {
209  output = true;
210  }
211  });
212  visitDepthFirst(prog, [&](const RamLoad& load) {
213  if (load.getRelation() == rel) {
214  input = true;
215  }
216  });
217  addRelation(rel.getName(), interface, input, output);
218  id++;
219  }
220  }
221  ~RAMIProgInterface() override {
222  for (auto* interface : interfaces) {
223  delete interface;
224  }
225  }
