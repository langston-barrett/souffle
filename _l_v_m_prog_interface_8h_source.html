172  * Implementation of SouffleProgram interface for an interpreter instance

174 class LVMProgInterface : public SouffleProgram {
175 public:
176  LVMProgInterface(LVMInterface& interp)
177  : prog(*interp.getTranslationUnit().getProgram()), exec(interp),
178  symTable(interp.getTranslationUnit().getSymbolTable()) {
179  uint32_t id = 0;
180 
181  // Retrieve AST Relations and store them in a map
182  std::map<std::string, const RamRelation*> map;
183  visitDepthFirst(prog, [&](const RamRelation& rel) { map[rel.getName()] = &rel; });
184 
185  // Build wrapper relations for Souffle's interface
186  for (auto& relPtr : exec.getRelationMap()) {
187  if (relPtr == nullptr) {
188  continue;
189  }
190  const auto& name = relPtr->getName();
191  auto& interpreterRel = *relPtr;
192  assert(map[name]);
193  const RamRelation& rel = *map[name];
194 
195  // construct types and names vectors
196  std::vector<std::string> types;
197  std::vector<std::string> attrNames;
198  for (size_t i = 0; i < rel.getArity(); i++) {
199  std::string t = rel.getArgTypeQualifier(i);
200  types.push_back(t);
201 
202  std::string n = rel.getArg(i);
203  attrNames.push_back(n);
204  }
205  auto* interface =
206  new LVMRelInterface(interpreterRel, symTable, rel.getName(), types, attrNames, id);
207  interfaces.push_back(interface);
208  bool input;
209  bool output;
210  visitDepthFirst(prog, [&](const RamStore& store) {
211  if (store.getRelation() == rel) {
212  output = true;
213  }
214  });
215  visitDepthFirst(prog, [&](const RamLoad& load) {
216  if (load.getRelation() == rel) {
217  input = true;
218  }
219  });
220  addRelation(rel.getName(), interface, input, output);
221  id++;
222  }
223  }
224  ~LVMProgInterface() override {
225  for (auto* interface : interfaces) {
226  delete interface;
227  }
228  }
