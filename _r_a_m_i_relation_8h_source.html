85  /** Insert tuple */
86  virtual void insert(const RamDomain* tuple) {
87  assert(tuple);
88 
89  // make existence check
90  if (exists(tuple)) {
91  return;
92  }
93 
94  // check for null-arity
95  if (arity == 0) {
96  indices[0].insert(tuple);
97  num_tuples = 1;
98  return;
99  }
100 
101  int blockIndex = num_tuples / (BLOCK_SIZE / arity);
102  int tupleIndex = (num_tuples % (BLOCK_SIZE / arity)) * arity;
103 
104  if (tupleIndex == 0) {
105  blockList.push_back(std::make_unique<RamDomain[]>(BLOCK_SIZE));
106  }
107 
108  RamDomain* newTuple = &blockList[blockIndex][tupleIndex];
109  for (size_t i = 0; i < arity; ++i) {
110  newTuple[i] = tuple[i];
111  }
112 
113  // update all indexes with new tuple
114  for (auto& cur : indices) {
115  cur.insert(newTuple);
116  }
117 
118  // increment relation size
119  num_tuples++;
120  }

310  /** Find the new knowledge generated by inserting a tuple */
311  std::vector<RamDomain*> extend(const RamDomain* tuple) override {
312  std::vector<RamDomain*> newTuples;
313 
314  newTuples.push_back(new RamDomain[2]{tuple[0], tuple[0]});
315  newTuples.push_back(new RamDomain[2]{tuple[0], tuple[1]});
316  newTuples.push_back(new RamDomain[2]{tuple[1], tuple[0]});
317  newTuples.push_back(new RamDomain[2]{tuple[1], tuple[1]});
318 
319  std::vector<const RamDomain*> relevantStored;
320  for (const RamDomain* vals : *this) {
321  if (vals[0] == tuple[0] || vals[0] == tuple[1] || vals[1] == tuple[0] || vals[1] == tuple[1]) {
322  relevantStored.push_back(vals);
323  }
324  }
325 
326  for (const auto vals : relevantStored) {
327  newTuples.push_back(new RamDomain[2]{vals[0], tuple[0]});
328  newTuples.push_back(new RamDomain[2]{vals[0], tuple[1]});
329  newTuples.push_back(new RamDomain[2]{vals[1], tuple[0]});
330  newTuples.push_back(new RamDomain[2]{vals[1], tuple[1]});
331  newTuples.push_back(new RamDomain[2]{tuple[0], vals[0]});
332  newTuples.push_back(new RamDomain[2]{tuple[0], vals[1]});
333  newTuples.push_back(new RamDomain[2]{tuple[1], vals[0]});
334  newTuples.push_back(new RamDomain[2]{tuple[1], vals[1]});
335  }
336 
337  return newTuples;
338  }
339  /** Extend this relation with new knowledge generated by inserting all tuples from a relation */
340  void extend(const RAMIRelation& rel) override {
341  std::vector<RamDomain*> newTuples;
342  // store all values that will be implicitly relevant to the those that we will insert
343  for (const auto* tuple : rel) {
344  for (auto* newTuple : extend(tuple)) {
345  newTuples.push_back(newTuple);
346  }
347  }
